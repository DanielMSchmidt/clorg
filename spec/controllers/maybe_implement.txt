TESTS:
  describe "DELETE 'destroy_myself'" do
    before(:each) do
      @user = FactoryGirl.create(:user)
    end

    describe "success" do
      it "should delete yourself" do
        test_sign_in(@user)
        lambda do
          delete :destroy_myself, :id => @user
        end.should change(User, :count).by(-1)
      end
    end

    describe "failure" do
      before(:each) do
        @second = FactoryGirl.create(:user, :email => "another@email.de", :name => "another Name")
      end
      # change the tests, into the click_botton and so on scheme
      it "should not delete someone else" do
        test_sign_in(@user)
        lambda do
          delete :destroy_myself, :id => @second
        end.should_not change(User, :count)
      end

      it "should not delete an admin" do
        @user.toggle!(:admin)
        test_sign_in(@user)
        lambda do
          delete :destroy_myself, :id => @user
        end.should_not change(User, :count)
      end
    end
  end

Controller:
  def destroy_myself
    if current_user.admin?
      flash[:notice] = "Don't delete yourself as admin"
    else
      current_user.destroy
      redirect_to(root_path)
      sign_out
    end
  end
